### Project Choice Step (NEW)

```tsx
// Step 3: Team Setup - Create or Join Project
export function ProjectChoiceStep({ onNext, userId }) {
  const [choice, setChoice] = useState<'create' | 'join' | null>(null)
  const [joinCode, setJoinCode] = useState('')

  const handleCreateProject = () => {
    setChoice('create')
    onNext({ projectChoice: 'create' })
  }

  const handleJoinProject = async () => {
    if (!joinCode) return

    try {
      // Look up project by code or ID
      const project = await findProjectByCode(joinCode)

      if (!project) {
        alert('Project not found')
        return
      }

      // Request to join (can be auto-approved or require admin approval)
      await fetch(`/api/jam/projects/${project.id}/members`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ userId, role: 'MEMBER' }),
      })

      onNext({
        projectChoice: 'join',
        projectId: project.id,
        skipProjectCreation: true,
      })
    } catch (error) {
      console.error('Failed to join project:', error)
    }
  }

  return (
    <div className="space-y-6">
      <div>
        <h2 className="mb-2 text-2xl font-bold">Team Setup</h2>
        <p className="text-foreground">Create a new project or join an existing team</p>
      </div>

      <div className="grid gap-4 md:grid-cols-2">
        {/* Create New Project */}
        <Card
          className={`cursor-pointer transition-all ${
            choice === 'create' ? 'border-primary ring-primary ring-2' : ''
          }`}
          onClick={handleCreateProject}
        >
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              <Plus className="h-5 w-5" />
              Create New Project
            </CardTitle>
          </CardHeader>
          <CardContent>
            <p className="text-foreground text-sm">
              Start a new project as the team admin. You can invite members later.
            </p>
          </CardContent>
        </Card>

        {/* Join Existing Project */}
        <Card
          className={`cursor-pointer transition-all ${
            choice === 'join' ? 'border-primary ring-primary ring-2' : ''
          }`}
          onClick={() => setChoice('join')}
        >
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              <Users className="h-5 w-5" />
              Join Existing Project
            </CardTitle>
          </CardHeader>
          <CardContent>
            <p className="text-foreground mb-3 text-sm">Join a team that's already been created</p>
            {choice === 'join' && (
              <div className="space-y-2">
                <Input
                  value={joinCode}
                  onChange={(e) => setJoinCode(e.target.value)}
                  placeholder="Enter project code or ID"
                />
                <Button onClick={handleJoinProject} className="w-full">
                  Join Project
                </Button>
              </div>
            )}
          </CardContent>
        </Card>
      </div>

      <p className="text-foreground text-center text-xs">
        You can join multiple projects later from your dashboard
      </p>
    </div>
  )
}
```

export async function POST(request: Request) {
  const data = await request.json()
  const userId = await getUserId(request)

  // Save profile data
  await db.insert(profiles).values({
    userId,
    ...data,
  })

  // Mark onboarding complete
  await db.update(users).set({ onboardingComplete: true }).where(eq(users.id, userId))

  return NextResponse.json({ success: true })
}
