### Verification Dialog

```tsx
// src/components/jam-platform/admin/VerificationDialog.tsx
'use client'

import { useState } from 'react'
import { Button } from '@/components/ui/button'
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from '@/components/ui/dialog'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Textarea } from '@/components/ui/textarea'

export function VerificationDialog({
  submission,
  action,
  trigger,
}: {
  submission: any
  action: 'verify' | 'reject'
  trigger: React.ReactNode
}) {
  const [open, setOpen] = useState(false)
  const [notes, setNotes] = useState('')
  const [paymentTxHash, setPaymentTxHash] = useState('')
  const [loading, setLoading] = useState(false)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setLoading(true)

    try {
      const response = await fetch('/api/jam/admin/verify', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          projectQuestId: submission.id,
          action,
          verificationNotes: notes,
          paymentTxHash: action === 'verify' ? paymentTxHash : undefined,
        }),
      })

      if (response.ok) {
        setOpen(false)
        window.location.reload()
      } else {
        const error = await response.json()
        alert(error.message || 'Failed to process verification')
      }
    } catch (error) {
      console.error('Verification failed:', error)
    } finally {
      setLoading(false)
    }
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>{trigger}</DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>
            {action === 'verify' ? 'Verify Quest Submission' : 'Reject Quest Submission'}
          </DialogTitle>
          <DialogDescription>
            {action === 'verify'
              ? 'Approve this submission and optionally record payment details'
              : 'Reject this submission with feedback for the team'}
          </DialogDescription>
        </DialogHeader>

        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <Label htmlFor="notes">
              {action === 'verify' ? 'Verification Notes (Optional)' : 'Rejection Reason'}
            </Label>
            <Textarea
              id="notes"
              value={notes}
              onChange={(e) => setNotes(e.target.value)}
              placeholder={
                action === 'verify'
                  ? 'Great work! Meets all requirements...'
                  : 'Please address the following issues...'
              }
              required={action === 'reject'}
              rows={4}
            />
          </div>

          {action === 'verify' && (
            <div>
              <Label htmlFor="txHash">Payment Transaction Hash (Optional)</Label>
              <Input
                id="txHash"
                value={paymentTxHash}
                onChange={(e) => setPaymentTxHash(e.target.value)}
                placeholder="0x..."
              />
              <p className="text-foreground mt-1 text-xs">
                Record the on-chain payment transaction for bounty payout
              </p>
            </div>
          )}

          <div className="flex gap-2">
            <Button
              type="button"
              variant="outline"
              onClick={() => setOpen(false)}
              disabled={loading}
            >
              Cancel
            </Button>
            <Button
              type="submit"
              variant={action === 'verify' ? 'default' : 'destructive'}
              disabled={loading}
            >
              {loading
                ? 'Processing...'
                : action === 'verify'
                  ? 'Verify & Approve'
                  : 'Reject Submission'}
            </Button>
          </div>
        </form>
      </DialogContent>
    </Dialog>
  )
}
```

### Verification API

```typescript
// src/app/api/jam/admin/verify/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/db'
import { projectQuests } from '@/db/schema'
import { eq } from 'drizzle-orm'

export async function POST(request: NextRequest) {
  const { projectQuestId, action, verificationNotes, paymentTxHash } = await request.json()

  const currentUser = await getCurrentUser(request)

  // Check admin permissions
  const isAdmin = await checkAdminPermissions(currentUser.id)
  if (!isAdmin) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 403 })
  }

  // Update project quest
  const updateData: any = {
    verifiedBy: currentUser.id,
    verificationNotes,
    verifiedAt: new Date(),
    updatedAt: new Date(),
  }

  if (action === 'verify') {
    updateData.status = 'VERIFIED'
    updateData.isVerified = true
    if (paymentTxHash) {
      updateData.paymentTxHash = paymentTxHash
      updateData.paidAt = new Date()
    }
  } else {
    updateData.status = 'REJECTED'
    updateData.isVerified = false
  }

  await db.update(projectQuests).set(updateData).where(eq(projectQuests.id, projectQuestId))

  // TODO: Send notification to team

  return NextResponse.json({ success: true })
}
```
