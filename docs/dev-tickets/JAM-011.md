### Session History

```tsx
// src/app/jam/sessions/page.tsx
export default async function SessionsPage() {
  const sessions = await getUserSessions()

  return (
    <div className="container py-6">
      <h1 className="mb-6 text-3xl font-bold">Session History</h1>

      <div className="space-y-4">
        {sessions.map((session) => (
          <SessionCard key={session.id} session={session} />
        ))}
      </div>
    </div>
  )
}

function SessionCard({ session }) {
  return (
    <Card>
      <CardHeader>
        <div className="flex justify-between">
          <div>
            <CardTitle>Session with {session.mentor.displayName}</CardTitle>
            <p className="text-foreground text-sm">
              {formatDate(session.date)} ‚Ä¢ {session.duration} minutes
            </p>
          </div>
          <div className="flex items-center gap-1">
            <StarRating value={session.rating} readonly size="sm" />
          </div>
        </div>
      </CardHeader>

      <CardContent>
        <div className="space-y-3">
          <div>
            <h4 className="font-medium">Outcomes</h4>
            <p className="text-foreground text-sm">{session.outcomes}</p>
          </div>

          {session.nextSteps && (
            <div>
              <h4 className="font-medium">Next Steps</h4>
              <p className="text-foreground text-sm">{session.nextSteps}</p>
            </div>
          )}
        </div>
      </CardContent>
    </Card>
  )
}
```
